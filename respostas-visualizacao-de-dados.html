Exercício 1

starwars |>
  drop_na(sex) |> 
  ggplot(

Exercício 2

starwars |>
  drop_na(sex) |> 
  ggplot(
    aes(fct_infreq(sex), fill = sex)
    ) +
  geom_bar() +
  labs(
    x = "Sexo",y = "Contagem"
    ) +
  theme_classic() +
  theme(
    legend.position = "none"

gapminder |>
