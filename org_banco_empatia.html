library(tidyverse)
library(janitor)
library(haven)
library(readxl)
library(careless)
library(gt)
library(lavaan)
library(EGAnet)
df <- read_spss("data/empatia.sav") |> 
  clean_names()

itens <- read_excel("data/formulario_empatia.xlsx")  
df <- df |> 
  clean_names() |> 
  select(!qualsuacorraca:afetos_negativos) |>
  select(!c(qualseusexo, renda_per_capta)) |> 
  mutate(
    across(where(is.labelled), as_factor)
  ) |> 
  filter(idade >= 18)
df <- df |> 
  mutate(
    formacao_academica = case_when(str_detect(area_de_formacao, "Ciências Sociais, Jornalismo e Informação|Artes e Humanidades|Pedagogia, cursos de licenciatura") ~ "Ciências sociais, artes e educação",
                                   str_detect(area_de_formacao, "Odontologia") ~ "Saúde",
                                   .default = "Ciências exatas e biológicas")
  ) 
df |> write_csv("data/empatia_congresso.csv")

Escala de Racismo Moderno

racismo <- df |> 
  select(r1:r14, renda_familiar:formacao_academica, -area_de_formacao)
racismo |> 
  write_csv("data/dados_racismo_moderno.csv")
itens_racismo <- itens |> 
  select(`Eles têm conseguido mais do que merecem`:`Possuem uma beleza diferente`) |> 
  names() |> 
  tibble(Item = _) 
variaveis_racismo <- racismo |> 
  select(r1:r14) |> 
  names() |> 
  tibble(Variável = _)
fator_racismo <- tibble(
  Fator = c(rep("Negação do preconceito", 7),
            rep("Afirmação das diferenças", 3),
            rep("Negação do preconceito", 2),
            rep("Afirmação das diferenças", 2))
)
variaveis_racismo |> 
  bind_cols(itens_racismo) |> 
  bind_cols(fator_racismo) |> 
  write_csv("data/code_book_escala_racismo_moderno.csv")
